92  template <typename T>
93  void RemoveUnusedEntries(IndexVec& indices, std::vector<T>& entries) const {
94  // compute an inverse lookup index, i.e. for each
95  // vertex, store what indices point to it
96  std::vector<size_t> inverseIndex(entries.size(),0);
97  for (size_t i = 0;i<indices.size();++i) {
98  inverseIndex[indices[i]]++;
99  }
100 
101  for (int64_t i = int64_t(inverseIndex.size()-1);i>=0;--i) {
102  // unused vertex found, i.e. vertex that has no indices pointing to it
103  if (inverseIndex[size_t(i)] == 0) {
104  // shift all indices
105  for (size_t j = 0;j<indices.size();++j) {
106  if (indices[j] > size_t(i) ) indices[j]--;
107  }
108  // delete vertex
109  entries.erase(entries.begin()+size_t(i));
110  }
111  }
112  }
