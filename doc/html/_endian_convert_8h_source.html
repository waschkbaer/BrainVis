139  double AVG=0.0,VAR=0.0;
140  double AVG2=0.0,VAR2=0.0;

145  Swap<T>(&data[i]);
146  d=double(data[i]);
147  AVG2+=d;
148  VAR2+=d*d;
149  // Early NaN-Check to speed up the process
150  if ((i%1024)!=1023 && i!=NumElements-1) {}
151  else {
152  if (IsNaN<double>(VAR)) { // early out, swapped is correct
153  for (size_t j=i+1; j<NumElements; j++) {
154  Swap<T>(&data[j]);
155  d=double(data[j]);
156  AVG2+=d;
157  VAR2+=d*d;
158  }
159  AVG2/=double(NumElements);
160  VAR2=(VAR2-double(NumElements)*AVG2*AVG2)/double(NumElements-1);
